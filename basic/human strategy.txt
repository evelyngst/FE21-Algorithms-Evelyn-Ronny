GUESS = ROUND UP(UPPER LIMIT - LOWER LIMIT / 2)

DO
    IF GUESS > NUMBER 
        OLD GUESS = GUESS
        GUESS = ROUND UP(OLD GUESS - LOWER LIMIT / 2)
        UPPER LIMIT = OLD GUESS

    ELSE IF GUESS < NUMBER 
        OLD GUESS = GUESS
        GUESS = OLD GUESS + (ROUND UP(UPPER LIMIT - OLD GUESS) / 2))
        LOWER LIMIT = OLD GUESS

UNTIL GUESS == NUMBER
