DEF WEEK = 604800
DEF DAY = 86400
DEF HOUR = 3600
DEF MIN = 60

INPUT numberInSeconds

weeks = ROUND DOWN(numbersInSeconds / WEEK)
x = numberInSeconds % WEEK
days = ROUND DOWN (x / DAY)
x = x % DAY
hours = ROUND DOWN (x / HOUR)
x = x % HOUR
mins = ROUND DOWN (x / MIN)
sec = x % MINS

OUTPUT weeks "weeks" + days "days" + hours "hours" + mins "minutes" + sec "seconds"
